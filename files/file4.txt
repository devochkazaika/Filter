while((c=this.read())!=-1){
            letter = (char) c;
            if (letter == '\r') { continue;}
            if (letter == '\n'){
                text.key = (text.key ==0) ? 1 : text.key;
                switch (text.key) {
                    case 1:
                        System.out.println(text.value + "=" + "int");
                        integers.add(text.value);
                        break;
                    case 2:
                        System.out.println(text.value + "=" +"float");
                        floats.add(text.value);
                    case 3:
                        System.out.println(text.value + "=" +"str");
                        strings.add(text.value);
                }
                text.value = "";
                text.key = 0;
                index = 0;
            }
            else{
                if (!NumericLetters.contains(letter)){
                    text.key = 3;
                }
                else{
                    if (letter == '-' || letter == '+'){
                        //Если '+' стоит первым или '-' или '+' стоит последним тогда записать в string
                        c =this.read();
                        if (letter == '+' && index != 0 || (char) c == -1){
                            text.key = 3;
                        }
                        else{
                            //Если '-' НЕ первый или после него НЕ идет цифра записать в string
                            if (!((index == 0) || text.value.charAt(index+1) >= '0' && text.value.charAt(index+1) <= '9')){
                                text.key = 3;
                            }
                        }
                    }
                    else if (letter == 'E'){
                        //Если E НЕ последний и после него НЕ идет '-' или '+' записать в string
                        if (!((index != text.value.length()-1) && text.value.charAt(index+1) == '-' || text.value.charAt(index+1) == '+') || forE){
                            text.key = 3;
                        }
                        forE = true;
                    }
                    else if (letter == '.'){
                        if (index == 0 || text.value.charAt(index-1) > '9' || text.value.charAt(index-1) < '0' || forPoint){
                            text.key = 3;
                        }
                        else if (text.key < 2) text.key = 2;

                        forPoint = true;
                    }
                }
                text.value += letter;
                index+=1;
            }
        } 
    }
